class Tenant:
    def __init__(self, name):
        self.name = name

class TenantRepository:
    def __init__(self):
        self.tenants = []

    def create_tenant(self, name):
        tenant = Tenant(name)
        self.tenants.append(tenant)

    def get_tenant_by_name(self, name):
        for tenant in self.tenants:
            if tenant.name == name:
                return tenant
        return None
